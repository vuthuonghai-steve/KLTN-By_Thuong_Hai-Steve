 def load_pipeline_from_skills(pipeline_name: str, skills_yaml_path: str) -> dict:
     """Load pipeline definition từ skills.yaml thay vì hardcode"""
     with open(skills_yaml_path) as f:
         skills_data = yaml.safe_load(f)

     pipeline = skills_data['pipelines'].get(pipeline_name)
     if not pipeline:
         raise ValueError(f"Pipeline {pipeline_name} not found")

     return pipeline

 3.1 Thêm Pre-flight Phase

 ## Mandatory Pre-flight Phase (BẮT BUỘC)

 Trước khi execute skill, Agent PHẢI:

 1. **Scan skill directory structure**
    - List all subdirectories: knowledge/, loop/, scripts/, templates/, data/

 2. **Load Knowledge Resources**
    - Read all files in knowledge/ folder
    - Inject vào context trước khi xử lý task

 3. **Load Loop/Verification**
    - Read loop/checklist.md nếu tồn tại
    - Load validation scripts từ scripts/

 4. **Context Assembly**
    - Combine: SKILL.md + knowledge/* + loop/* → Full Context
    - Pass to main execution phase

 3.2 Thêm Guardrail

 ## Guardrails (Bổ sung)

 G6: **Enforce Knowledge Loading**
 - Trước khi bắt đầu work, phải gọi ls/glob kiểm tra knowledge/ tồn tại
 - Phải Read ít nhất 1 file từ knowledge/ trước khi output
 - Nếu knowledge/ trống → warning nhưng vẫn chạy
 - Nếu knowledge/ có file mà không đọc → GUARDRAIL VIOLATION

 Output

 - Skill-executor bắt buộc scan và load knowledge resources

 4.2 Cập nhật spawn_skill.py

 def prepare_task_with_awareness(pipeline_config: dict, stage: dict, queue_state: dict) -> dict:
     """Build task input với đầy đủ pipeline awareness"""

     predecessors = []
     for dep_id in stage.get('depends_on', []):
         if dep_id in queue_state['stages']:
             predecessors.append({
                 'stage_id': dep_id,
                 'skill': queue_state['stages'][dep_id]['skill'],
                 'output_dir': queue_state['stages'][dep_id].get('output_dir')
             })

     return {
         'pipeline_context': {
             'pipeline_id': queue_state['pipeline_id'],
             'pipeline_name': pipeline_config['pipeline_name'],
             'current_stage_index': len(queue_state['stages']),
             'total_stages': len(pipeline_config['stages']),
             'predecessors': predecessors,
             'global_context': pipeline_config.get('global_context', {})
         }
         # ... rest of task
     }
